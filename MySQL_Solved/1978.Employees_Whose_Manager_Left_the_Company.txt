# The desciption says that when a manager leave the company it gets deleted from the table however the report only stays in the manger_id so that is a big clue.


#We want to select the employees 
SELECT employee_id 

#This specifies the table we are retrieving it from.
FROM Employees 

#This we filter out the salary then check the manger_id that are not in the employee_id column implying they left the company.
WHERE salary < 30000 AND manager_ID not in (select employee_id from Employees) 

#So finally we order it by employee_id and its already makes in asc order.
ORDER BY employee_id
